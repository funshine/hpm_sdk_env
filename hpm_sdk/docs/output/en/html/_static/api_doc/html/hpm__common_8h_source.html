284 {
285  uint32_t i = 0;

304 static inline uint32_t get_first_set_bit_from_msb(uint32_t value)
305 {
306  uint32_t i = 31;
307  if (!value) {
308  return 0xFFFFFFFFUL;
309  }

314  }
